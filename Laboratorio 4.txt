int TRIG = 10;			// trigger en pin 10
int ECO = 9;			// echo en pin 9
int LED = 3;			// LED en pin 3
int DURACION;
int DISTANCIA;

void setup()
{
  pinMode(TRIG, OUTPUT); 	// trigger como salida
  pinMode(ECO, INPUT);		// echo como entrada
  pinMode(LED, OUTPUT);		// LED como salida
  Serial.begin(9600);  		// inicializacion de comunicacion serial a 9600 bps, "Serial" es el comando usado para mostrar en pantalla del computador la conexion de el arduino al computador, 9600 es la velocidad de la tasa de comunicacion

}

void loop()
{
  
  digitalWrite(TRIG, HIGH); 		// generacion del pulso a enviar
  delay(1);				// al pin conectado al trigger
  digitalWrite(TRIG, LOW);		// del sensor
  
  DURACION = pulseIn(ECO, HIGH);	// con funcion pulseIn se espera un pulso
  					// alto en Echo
  DISTANCIA = DURACION / 58.2;		// distancia medida en centimetros, el 58.2 es un valor de fabrica por el sensor para calcular en cm
  Serial.println(DISTANCIA);		// imprime el valor de la distancia que se calculo en DISTANCIA
  delay(200);				// demora entre datos
  // este if evalua verdadero si la distancia esta entre 0 y 20 centimetros 
  if (DISTANCIA <= 20 && DISTANCIA >=0 ) {
    digitalWrite (LED, HIGH);
    delay(DISTANCIA * 10);       // este delay en vez de ser un valor fijo, es el producto de la distancia * 10, haciendo que al medir la distancia por ejemplo en 50cm el delay arrojara 50*10 es decir 500ms, si el objeto se sigue acercando entonces el delay sera un numero menor haciendo que el led parpadee mas rapido 
    digitalWrite(LED, LOW);
  }


   
}
